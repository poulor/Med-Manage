// Database design  

Table Users as U {   

  user_id int [pk, increment] // auto-increment   

  first_name varchar
  
  middle_name varchar
  
  last_name varchar
  
  gender varchar

  username varchar   

  password varchar   

  user_type int   

}   

Table Doctors as U {   

  doctor_id int [pk, increment] // auto-increment   

  phone_number phone_number   

  specialty varchar

  department varchar   

  user_id int [pk, increment]   

}   

Table Patients as U {   

  patient_id int [pk, increment] // auto-increment   

  age int   

  date_of_birth date   

  height double   

  weight double   

  address varchar   

  phone_number varchar   

  health_history varchar
  
  notes varchar
  
  user_id int [pk, increment] 

}   

Table Receptionists as U {   

  receptionist_id int [pk, increment] // auto-increment   

  join_date date   
  
  user_id int [pk, increment] 

}   

Table Appointments as U {   

  appointment_id int [pk, increment] // auto-increment   

  appointment_date date   

  appointment_time timestamp  

  notes varchar  

  doctor_id int [pk, increment]   

  patient_id int [pk, increment]   

}   

Table Administrators as U {

  administrator_id int [pk, increment] // auto-increment   

  join_date date 
  
  user_id int [pk,increment]

} 

Ref: Patients.patient_id < Appointments.patient_id   

Ref: Doctors.doctor_id < Appointments.doctor_id   

Ref: Doctors.user_id < Patients.patient_id

Ref: Users.user_id - Doctors.user_id

Ref: Users.user_id - Administrators.user_id

Ref: Users.user_id - Patients.user_id

Ref: Users.user_id - Receptionists.user_id
